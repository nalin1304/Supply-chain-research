\paragraph{Cost objective.}
\begin{equation}
Z_1(\mathbf{x}) \;=\; \sum_{w \in W} \sum_{c \in C}
\sum_{v \in V} \;

\label{eq:cost-objective}

\paragraph{Emission objective.}
\begin{equation}
Z_2(\mathbf{x}) \;=\; \sum_{w \in W} \sum_{c \in C}
\sum_{v \in V} \;
\bigl( k_v + L_v \, x_{wcv} \bigr) \,
\cdot \, 2 \, d_{wc},
\label{eq:emission-objective}
\end{equation}

\paragraph{Demand fulfilment.}
\begin{equation}
\sum_{w \in W} \sum_{v \in V} x_{wcv} \;=\; D_c
\quad \forall c \in C,
\label{eq:demand-constraint}
\end{equation}

\paragraph{Warehouse capacity.}
\begin{equation}
\sum_{c \in C} \sum_{v \in V} x_{wcv} \;\leq\; S_w
\quad \forall w \in W,
\label{eq:capacity-constraint}
\end{equation}

\paragraph{Non-negativity.}
\begin{equation}
x_{wcv} \;\geq\; 0
\quad \forall w \in W, c \in C, v \in V.
\label{eq:nonnegativity}
\end{equation}

\paragraph{Volume-weighted delivery time.}
\begin{equation}
f_3(\mathbf{x}) \;=\;
\frac{

\displaystyle
\sum_{w \in W} \sum_{c \in C} \sum_{v \in V}

\end{equation}
